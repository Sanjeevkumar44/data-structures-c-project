#include <stdio.h>
#include <stdlib.h>
#define MAX_CUSTOMERS 100
#define MAX_AGENTS 5
typedef struct {
 int customerID;
} Customer;
typedef struct {
 Customer customers[MAX_CUSTOMERS];
 int front, rear, size;
} Queue;
void initQueue(Queue *q) {
 q->front = 0;
 q->rear = -1;
 q->size = 0;
}
int isFull(Queue *q) {
 return q->size == MAX_CUSTOMERS;
}
int isEmpty(Queue *q) {
 return q->size == 0;
}
void enqueue(Queue *q, int customerID) {
 if (isFull(q)) {
 printf("Queue is full. Cannot add more customers.\n");
 return;
 }
 q->rear = (q->rear + 1) % MAX_CUSTOMERS;
 q->customers[q->rear].customerID = customerID;
 q->size++;
}
int dequeue(Queue *q) {
 if (isEmpty(q)) {
 printf("Queue is empty. No customers to serve.\n");
 return -1;
 }
 int customerID = q->customers[q->front].customerID;
 q->front = (q->front + 1) % MAX_CUSTOMERS;
 q->size--;
 return customerID;
}
int main() {
 Queue customerQueue;
 initQueue(&customerQueue);
 int agents[MAX_AGENTS] = {0}; 
 int customerID = 1;
 char command;
 while (1) {
 printf("Enter 'a' to add customer, 's' to serve customer, 'q' to quit: ");
 scanf(" %c", &command);
 if (command == 'a') {
 enqueue(&customerQueue, customerID++);
 } else if (command == 's') {
 int served = 0;
 for (int i = 0; i < MAX_AGENTS; i++) {
 if (agents[i] == 0) {
 int customer = dequeue(&customerQueue);
 if (customer != -1) {
 agents[i] = 1;
 printf("Agent %d is serving Customer %d\n", i + 1, customer);
 served = 1;
 agents[i] = 0; 
 }
 break;
 }
 }
 if (!served) {
 printf("All agents are busy. Please wait.\n");
 }
 } else if (command == 'q') {
 break;
 } else {
 printf("Invalid command.\n");
 }
 }
 return 0;
}
